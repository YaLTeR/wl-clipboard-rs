Unloads the kernel module with the given name.

Example

use std::ffi::CString;
use nix::kmod::{delete_module, DeleteModuleFlags};

delete_module(&CString::new("mymod").unwrap(), DeleteModuleFlags::O_NONBLOCK).unwrap();
